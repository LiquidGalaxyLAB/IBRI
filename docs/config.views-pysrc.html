 49      Download File returns an KML filename as petition. The files have the 
 50      following format: IBRI[m]R[r].kml where the m is the mission id and the 
 51      r is the route id. 
 52   
 53      @param request: Request petition that is handled by Django 
 54      @param m: Represents the mission id 
 55      @param r: Represents the route id 
 56      @see FileWrapper 
 57      @see Content-Disposition 
 58      """ 
 59   
 60      filename = os.path.join(KML_DIR, 'IBRI' + m, 'IBRI' + m + 'R' + r + '.kml') 
 61      wrapper = FileWrapper(file(filename)) 
 62      ct = 'application/vnd.google-earth.kml+xml' 
 63      response = HttpResponse(wrapper, content_type=ct) 
 64      cd = 'attachment; filename=%s' % os.path.basename(filename) 
 65      response['Content-Disposition'] = cd 
 66      response['Content-Length'] = os.path.getsize(filename) 
 67      return response
